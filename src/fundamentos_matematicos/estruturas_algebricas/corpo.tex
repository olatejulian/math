    \begin{defi}[Corpo]\label{defi:corpo}
        Um anel comutativo $F$ \ref{defi:anel-comutativo} é denominado corpo se satisfaz a condição: $(\forall a \in F\ \backslash\ \{0\})(\exists! b \in F): a\cdot b=\Set{I}$
    \end{defi}
